@name missile track projector
@inputs Calculate Tpos:vector Tvel:vector 
@outputs Impact:vector ImpactDist Go Done Error TTI ETTI Time Timer Accel:vector IntAcc:vector Inter:vector
@persist Vel:vector Acc:vector InputLoaded Go Timer Done Error IntAcc:vector Int
@trigger Calculate

Gl = entity():pos():z()
TTI = (Tpos:z()-Gl)/(-1*Tvel:z())
Accel=vec($Tvel:x(),$Tvel:y(),$Tvel:z())
Int+=1
IntAcc+=Accel
if (Int==6000) {Int=1
IntAcc=IntAcc/6000
Inter=IntAcc}


if (Calculate & !Go) {
    ETTI = TTI
    Timer=0
    Impact=Tpos
    Acc=Accel
    Vel=Tvel
    InputLoaded=1
    Go=1
    Done=0
    Error=0
}

if (Go==1) {
    runOnTick(1)
    Impact = Impact+(Vel)
    Vel=Vel+Acc
    Timer=Timer+1
    if (Impact:z()<=Gl) {
        Done=1
        Time=Timer
        Go=0
        InputLoaded=0
        ImpactDist = entity():pos():distance(Impact)
    }

    if (Timer>1500 | Vel:z()>0) {
        Error=1
        Go=0
        Done=1}
    
}
else {interval(1000)}
