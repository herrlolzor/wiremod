@name Supertable
@inputs Data X Y Z Clk Reset
@outputs Out Size
@persist A:table Format
@trigger Clk X Y Z
if (Clk==1) {
    runOnTick(1)
    B=A[X,table]
    C=A[X,table][Y,table]
    C[Z,number]=Data
    B[Y,table]=C
    A[X,table]=B
    }
if (Format==0) {Format=1
A=table(table(table()))}
Out=A[X,table][Y,table][Z,number]
Size=A:count()
